## 1. Image Management

# Purpose: Manage Docker images (pull, inspect, tag, save/load, remove)

docker pull nginx:alpine # Pull nginx image
docker pull redis:7-alpine # Pull redis image
docker pull node:24-alpine # Pull Node.js image
docker images # List all images
docker inspect nginx:alpine # Inspect nginx image details
docker tag nginx:alpine my-nginx:latest # Tag image with custom name
docker save -o nginx-alpine.tar nginx:alpine # Save image to tar file
docker load -i nginx-alpine.tar # Load image from tar
docker rmi my-nginx:latest # Remove tagged image
docker system prune -f # Cleanup unused images

## 2. Container Lifecycle

# Purpose: Run, manage, and interact with containers

docker run -d -p 8080:80 nginx:alpine # Run nginx container in detached mode
docker run -d --name my-redis --restart always redis:7-alpine # Run redis with restart policy
docker create --name my-node node:24-alpine # Create node container without starting
docker start my-node # Start the node container
docker exec -it my-node sh # Enter shell in node container
docker exec -it my-redis redis-cli # Open Redis CLI inside container
docker cp ./localfile.txt my-node:/app/ # Copy file from host to container
docker cp my-node:/app/file.txt ./hostfile.txt # Copy file from container to host

## 3. Monitoring and Troubleshooting

# Purpose: Observe containers and inspect details

docker stats # Monitor container resources
docker logs -f --timestamps nginx # Follow logs with timestamps
docker inspect my-node # Inspect container configuration
docker volume create my-volume # Create Docker volume
docker run -v my-volume:/data redis # Attach volume to container

## 4. Cleanup and Maintenance

# Purpose: Stop containers, remove resources, clean system

docker stop $(docker ps -q) # Stop all running containers
docker rm $(docker ps -a -q) # Remove all containers
docker rmi $(docker images -q) # Remove all images
docker volume rm $(docker volume ls -q) # Remove all volumes
docker network prune -f # Remove unused networks
